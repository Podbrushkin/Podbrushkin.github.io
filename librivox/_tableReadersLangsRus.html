// readers who have ever recorded in specific lang
MATCH (r:Reader)--(s:Section)--(b:Book)
WITH r,
    CASE 
        WHEN b.language = 'Multilingual' THEN s.language
        ELSE b.language
    END AS lang,
    COUNT(*) AS langCount
ORDER BY r.id, langCount DESC
WITH r, COLLECT({lang: lang, langCount: langCount}) AS langs
WHERE ANY(langObj IN langs WHERE langObj.lang = 'Russian')
UNWIND langs AS langObj
RETURN 'https://librivox.org/reader/'+r.id AS url, r.name AS reader, langObj.lang AS lang, langObj.langCount AS langCount
