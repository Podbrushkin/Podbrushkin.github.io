// All 2024 readers, with language and totalseconds
MATCH (r:Reader)--(s:Section)--(b:Book)
WHERE b.dateUploaded.year = 2024
RETURN 'https://librivox.org/reader/'+r.id AS url,r.name AS reader,
    CASE 
        WHEN b.language = 'Multilingual' THEN s.language
        ELSE b.language
    END AS lang,
    SUM(s.playtime) AS playtime
ORDER BY lang, playtime DESC
