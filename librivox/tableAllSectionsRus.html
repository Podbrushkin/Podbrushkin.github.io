MATCH (r:Reader)--(s:Section)--(b:Book)
OPTIONAL MATCH (a:Author)-[:AUTHOR]->(b)
WITH COLLECT(a)[0] AS a,b,r,s,
    CASE 
        WHEN b.language = 'Multilingual' THEN s.language
        ELSE b.language
    END AS lang
WHERE lang = 'Russian'
RETURN 
	// book
	b.urlLibrivox AS bookLv, 
	'http://archive.org/details/'+b.idIarchive AS bookIA, 
	b.language AS bookLang,
	date(b.dateUploaded) AS bookDate,
	COALESCE(a.firstName, '') + ' ' + COALESCE(a.lastName, '') AS bookAuthor,
	b.title AS bookTitle, 
	b.copyrightYear AS bookCopyrightYear,
	b.views AS bookViews,
	//section
	s.title AS sectionTitle,
	s.fileName AS sectionFile,
	// reader
	'https://librivox.org/reader/'+r.id AS readerLv,
	r.name AS readerName
ORDER BY b.id, sectionFile
